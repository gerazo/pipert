188 void SchedulerImpl::RunTasks() {
189  ThreadId::TagCurrentThread();
190  while (keep_running_.load(std::memory_order_acquire)) {
191  void* state = nullptr;
192  ChannelImpl* channel = nullptr;
193  PacketBase* packet = nullptr;
194  {
195  std::unique_lock<AdaptiveSpinLock> lock(global_mutex_);
196  while (keep_running_.load(std::memory_order_acquire) &&
197  state_queue_.empty()) {
198  global_covar_.wait(lock);
199  }
200  if (!state_queue_.empty()) {
201  state = state_queue_.front();
202  assert(state2channel_queues_.count(state) == 1);
203  auto& chq = state2channel_queues_[state];
204  assert(chq.enabled);
205  assert(!chq.heap.empty());
206  channel = chq.heap.front();
207  assert(channel->GetState() == state);
208  // state == nullptr has the stateless channels, so disabling is needed
209  if (state != nullptr) {
210  // set stateful channels for this state to disabled
211  chq.enabled = false;
212  state_queue_.pop_heap();
213  }
214  packet = channel->PopNext();
215  assert(packet);
216  }
217  }
218  if (packet) {
219  assert(channel);
220  ThreadId::SetNameOfCurrentThread(channel->GetName());
221  channel->Execute(packet);
222  if (state != nullptr) {
223  // set stateful channels for this state to enabled again
224  std::lock_guard<AdaptiveSpinLock> lock(global_mutex_);
225  assert(state_queue_.count(state) == 0);
226  assert(state2channel_queues_.count(state) == 1);
227  auto& chq = state2channel_queues_[state];
228  assert(!chq.enabled);
229  chq.enabled = true;
230  if (!chq.heap.empty()) {
231  state_queue_.push_heap(state);
232  // we do not need global_covar_.notify_one() here as this thread goes
233  // on by servicing this state
234  }
235  }
236  }
237  }
